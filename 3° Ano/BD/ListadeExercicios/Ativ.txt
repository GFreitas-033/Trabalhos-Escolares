SELECT nome
FROM Departamento
WHERE id_departamento = (
    SELECT id_departamento
    FROM Projeto
    WHERE (
        SELECT SUM(custo) AS custoRH
        FROM Projeto
        WHERE id_departamento = 10
    ) > ALL(
        SELECT SUM(custo) AS totalCusto
        FROM Projeto
        GROUP BY id_departamento
    )
);
/*Funcional*/
SELECT nome
FROM Departamento
WHERE id_departamento = (
    SELECT id_departamento
    FROM Projeto
    GROUP BY id_departamento
    HAVING SUM(custo) = (
        SELECT MAX(totalCusto)
        FROM (
            SELECT SUM(custo) AS totalCusto
            FROM Projeto
            GROUP BY id_departamento
        ) AS maxCusto
    )
);
